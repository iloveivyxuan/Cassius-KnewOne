Programming Evolution


/* C */
int fact( int n ) {
    if (n <= 1) 
        return 1;
    else
        return n * fact(n - 1);    
}
                    

;;Lisp
(define (fact n)
  (if (= n 0)
      1
    (* n (fact (- n 1)))))
                    

#Python
def fact(n):
    return reduce(lambda x, y: 
                      x * y, 
                      range(1, n + 1))
                    

#Ruby
def fact(n)
  (1..n).reduce(:*)
end
