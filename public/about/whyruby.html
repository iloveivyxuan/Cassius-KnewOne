Java Factorial


Public class Fact
{
    public static long fact(int n) {
        if(n <= 1)
            return 1;
        else
            return n * fact(n - 1);
    }

    public static void main(String[] args) {
        long result = Fact.fact(10);
        System.out.println(result);
    }
}
